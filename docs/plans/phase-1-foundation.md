### Task 1.3: Rewrite agent/db/schema.ts — refined 6 tables

**Files:**
- Modify: `agent/db/schema.ts` (full rewrite)
- Create: `agent/lib/db/schema.ts` (re-export)

**Key changes from old schema:**
- Remove `seniorityEnum` → contacts use plain `text` for seniority
- Remove `taskTypeEnum` → tasks use plain `text` for type
- Add `properties` JSONB on contacts: `[{key, value, type}]`
- Add `properties` JSONB on organisations: `[{key, value, type}]`
- Add `organisationMd` text on organisations (the "Soul.md")
- Add `icpSourceFiles` JSONB on organisations
- Add `honchoPeerId` on organisations
- Keep only 3 enums: `orgStatusEnum`, `runStatusEnum`, `taskStatusEnum`
- 6 tables: workspaces, organisations, agent_runs, contacts, signals, tasks

```ts
/**
 * RevenueOS — Database Schema (6 Tables, 3 Enums)
 *
 * Enums removed: seniorityEnum, taskTypeEnum → agent decides types dynamically
 * Enums kept: orgStatusEnum, runStatusEnum, taskStatusEnum (system-level)
 * New fields: properties JSONB on contacts + organisations, organisationMd on organisations
 */

import {
  pgEnum,
  pgTable,
  uuid,
  text,
  varchar,
  integer,
  timestamp,
  jsonb,
  index,
  uniqueIndex,
  type AnyPgColumn,
} from "drizzle-orm/pg-core";
import { relations } from "drizzle-orm";
import { sql } from "drizzle-orm/sql";

// ── Enums (3 — system-level only) ──────────────────────────────────

export const orgStatusEnum = pgEnum("org_status", [
  "onboarding",
  "active",
  "paused",
  "churned",
]);

export const runStatusEnum = pgEnum("run_status", [
  "pending",
  "running",
  "completed",
  "failed",
]);

export const taskStatusEnum = pgEnum("task_status", [
  "pending",
  "in_progress",
  "completed",
  "skipped",
  "failed",
]);

// ── Property type ──────────────────────────────────────────────────

type Property = { key: string; value: string; type: "text" | "number" | "url" | "date" };

// ── Helpers ────────────────────────────────────────────────────────

const now = () => timestamp({ withTimezone: true }).defaultNow().notNull();
const nullableTz = () => timestamp({ withTimezone: true });

// ── Tables ─────────────────────────────────────────────────────────

/** 1. workspaces — multi-tenant scope, Cognito sub */
export const workspaces = pgTable("workspaces", {
  id: uuid("id").defaultRandom().primaryKey(),
  cognitoSub: text("cognito_sub").notNull().unique(),
  name: text("name"),
  email: varchar("email", { length: 320 }).notNull(),
  avatarUrl: text("avatar_url"),
  createdAt: now(),
}, (t) => [
  index("workspaces_email_idx").on(t.email),
]);

/** 2. organisations — target account + agent identity */
export const organisations = pgTable("organisations", {
  id: uuid("id").defaultRandom().primaryKey(),
  workspaceId: uuid("workspace_id").notNull()
    .references(() => workspaces.id, { onDelete: "cascade" }),
  name: text("name").notNull(),
  domain: text("domain").notNull(),
  hqCountry: text("hq_country").notNull(),
  /** Free-text ICP description — fallback if no Organisation.md yet */
  icpDescription: text("icp_description").notNull(),
  /** The "Soul.md" — generated by initialisation agent from uploaded docs */
  organisationMd: text("organisation_md"),
  /** References to S3 files used to build ICP */
  icpSourceFiles: jsonb("icp_source_files").$type<Array<{
    url: string; filename: string; uploadedAt: string; extractedText?: string;
  }>>().default([]),
  /** Dynamic key-value properties (industry, employee_count, funding_stage, etc.) */
  properties: jsonb("properties").$type<Property[]>().default([]),
  opportunityScore: integer("opportunity_score"),
  lastResearchedAt: nullableTz(),
  nextRunAt: nullableTz(),
  refreshIntervalDays: integer("refresh_interval_days"),
  honchoPeerId: text("honcho_peer_id").unique(),
  status: orgStatusEnum("status").default("onboarding"),
  onboardingMetadata: jsonb("onboarding_metadata")
    .$type<Record<string, unknown>>(),
  createdAt: now(),
  updatedAt: now(),
}, (t) => [
  uniqueIndex("orgs_domain_workspace_idx").on(t.domain, t.workspaceId),
  index("orgs_workspace_id_idx").on(t.workspaceId),
  index("orgs_status_idx").on(t.status),
  index("orgs_next_run_at_idx").on(t.nextRunAt)
    .where(sql`${t.nextRunAt} IS NOT NULL`),
]);

/** 3. agent_runs — execution history */
export const agentRuns = pgTable("agent_runs", {
  id: uuid("id").defaultRandom().primaryKey(),
  orgId: uuid("org_id").notNull()
    .references(() => organisations.id, { onDelete: "cascade" }),
  ashSessionId: text("ash_session_id"),
  status: runStatusEnum("status").default("pending"),
  toolsInvoked: integer("tools_invoked").default(0),
  durationMs: integer("duration_ms"),
  tokensUsed: integer("tokens_used"),
  summary: text("summary"),
  icpFitScore: integer("icp_fit_score"),
  recommendedActions: jsonb("recommended_actions").$type<Array<{
    action: string; priority: number; rationale: string;
  }>>(),
  errorDetails: jsonb("error_details").$type<{
    message: string; toolName?: string; stackTrace?: string;
  }>(),
  startedAt: nullableTz(),
  completedAt: nullableTz(),
  createdAt: now(),
}, (t) => [
  index("runs_org_created_idx").on(t.orgId, t.createdAt),
  index("runs_status_idx").on(t.status),
]);

/** 4. contacts — discovered people */
export const contacts = pgTable("contacts", {
  id: uuid("id").defaultRandom().primaryKey(),
  orgId: uuid("org_id").notNull()
    .references(() => organisations.id, { onDelete: "cascade" }),
  agentRunId: uuid("agent_run_id")
    .references((): AnyPgColumn => agentRuns.id, { onDelete: "set null" }),
  name: text("name").notNull(),
  title: text("title"),
  linkedinUrl: text("linkedin_url"),
  email: text("email"),
  /** Dynamic — agent decides (e.g. "C-Suite", "VP", "Director", "Head of Engineering") */
  seniority: text("seniority").default("unknown"),
  /** Dynamic key-value properties (location, department, tenure, etc.) */
  properties: jsonb("properties").$type<Property[]>().default([]),
  source: text("source").default("exa"),
  relevanceNote: text("relevance_note"),
  discoveredAt: now(),
}, (t) => [
  uniqueIndex("contacts_org_linkedin_idx").on(t.orgId, t.linkedinUrl),
  index("contacts_org_id_idx").on(t.orgId),
]);

/** 5. signals — dynamic intelligence events */
export const signals = pgTable("signals", {
  id: uuid("id").defaultRandom().primaryKey(),
  orgId: uuid("org_id").notNull()
    .references(() => organisations.id, { onDelete: "cascade" }),
  agentRunId: uuid("agent_run_id")
    .references((): AnyPgColumn => agentRuns.id, { onDelete: "set null" }),
  contactId: uuid("contact_id")
    .references(() => contacts.id, { onDelete: "set null" }),
  /** Dynamic — agent decides (e.g. "funding_round", "leadership_change", "product_launch") */
  type: text("type").notNull(),
  title: text("title").notNull(),
  quotes: jsonb("quotes").notNull()
    .$type<Array<{ text: string; speaker?: string; source?: string }>>(),
  icpRelevance: text("icp_relevance").notNull(),
  sources: jsonb("sources").notNull()
    .$type<Array<{ url: string; title: string; publishedDate?: string }>>(),
  impact: integer("impact"),
  createdAt: now(),
}, (t) => [
  index("signals_org_id_idx").on(t.orgId),
]);

/** 6. tasks — engagement actions */
export const tasks = pgTable("tasks", {
  id: uuid("id").defaultRandom().primaryKey(),
  orgId: uuid("org_id").notNull()
    .references(() => organisations.id, { onDelete: "cascade" }),
  agentRunId: uuid("agent_run_id")
    .references((): AnyPgColumn => agentRuns.id, { onDelete: "set null" }),
  contactId: uuid("contact_id")
    .references(() => contacts.id, { onDelete: "set null" }),
  /** Dynamic — agent decides (e.g. "send_email", "linkedin_dm", "schedule_call", "research_deeper") */
  type: text("type").notNull(),
  status: taskStatusEnum("status").default("pending"),
  description: text("description").notNull(),
  payload: jsonb("payload").$type<Record<string, unknown>>(),
  rationale: text("rationale"),
  priority: integer("priority").default(50),
  executedAt: nullableTz(),
  result: text("result"),
  createdAt: now(),
  updatedAt: now(),
}, (t) => [
  index("tasks_org_id_idx").on(t.orgId),
  index("tasks_status_idx").on(t.orgId, t.status),
]);

// ── Relations ──────────────────────────────────────────────────────

export const workspacesRelations = relations(workspaces, ({ many }) => ({
  organisations: many(organisations),
}));

export const organisationsRelations = relations(organisations, ({ one, many }) => ({
  workspace: one(workspaces, {
    fields: [organisations.workspaceId], references: [workspaces.id],
  }),
  agentRuns: many(agentRuns),
  contacts: many(contacts),
  signals: many(signals),
  tasks: many(tasks),
}));

export const agentRunsRelations = relations(agentRuns, ({ one, many }) => ({
  organisation: one(organisations, {
    fields: [agentRuns.orgId], references: [organisations.id],
  }),
  contacts: many(contacts),
  signals: many(signals),
  tasks: many(tasks),
}));

export const contactsRelations = relations(contacts, ({ one, many }) => ({
  organisation: one(organisations, {
    fields: [contacts.orgId], references: [organisations.id],
  }),
  agentRun: one(agentRuns, {
    fields: [contacts.agentRunId], references: [agentRuns.id],
  }),
  signals: many(signals),
  tasks: many(tasks),
}));

export const signalsRelations = relations(signals, ({ one }) => ({
  organisation: one(organisations, {
    fields: [signals.orgId], references: [organisations.id],
  }),
  agentRun: one(agentRuns, {
    fields: [signals.agentRunId], references: [agentRuns.id],
  }),
  contact: one(contacts, {
    fields: [signals.contactId], references: [contacts.id],
  }),
}));

export const tasksRelations = relations(tasks, ({ one }) => ({
  organisation: one(organisations, {
    fields: [tasks.orgId], references: [organisations.id],
  }),
  agentRun: one(agentRuns, {
    fields: [tasks.agentRunId], references: [agentRuns.id],
  }),
  contact: one(contacts, {
    fields: [tasks.contactId], references: [contacts.id],
  }),
}));
```

### Task 1.5: Write agent/instructions.md — Agent DNA (always active)

**Files:**
- Modify: `agent/instructions.md`

This is the **base system prompt** — always active for every turn. It contains agent identity, research methodology, decision framework, memory protocol, and signal→task pipeline rules.

Dynamic context (Organisation.md, Honcho recall, stage context) is injected via `defineDynamic` instructions in Phase 3 — NOT via template strings here.

```markdown
# RevenueOS — Autonomous Account Research Agent

## Who You Are
You are an autonomous account research agent for APAC markets. You research target
companies on behalf of a seller. Your identity comes from the Organisation.md injected
into your context — this defines who your seller is, their ICP, and their ideal customer.

You are smart, strategic, and self-improving. Every run you get smarter because you
read what you learned last time and adjust your approach.

## What You MUST Do First
1. **Review your accumulated memory** (injected via Honcho recall)
   - What have you already tried? What worked? What didn't?
   - What contacts/signals/tasks already exist for this account?
2. **Review the account stage** (injected via stage context)
   - Is this a first run (new prospect)? An active deal? Re-engagement?
   - What research has already been done?
3. **Load relevant skills** based on:
   - Target company's HQ country → load the matching market skill
   - Account stage → load the matching stage skill

## Research Methodology
After reviewing context, decide your approach:
1. **For first-run / new prospects:** Deep research across all dimensions
   - Delegate to subagents for parallel research
   - Company-researcher: funding, expansion, regulatory, financial signals
   - People-researcher: stakeholders, roles, interests, events
2. **For active accounts:** Targeted refresh
   - Focus on what changed since last run
   - Check for new signals at known contacts
   - Look for trigger events
3. **For re-engagement:** Find new angles
   - Look for changes in leadership, strategy, or market conditions
   - Identify new stakeholders who joined

## Delegating to Subagents
You have two specialist subagents:
- **company-researcher:** Researches company-level signals (funding, expansion, regulatory, product launches). Has its own tools: exa_company_deep_dive, exa_answer, db_write_signals.
- **people-researcher:** Discovers and deeply profiles stakeholders (priorities, speeches, interests, events, social). Has its own tools: exa_people_search, exa_person_deep_dive, exa_answer, db_write_contacts.

Delegate when you need breadth. You synthesize their findings into decisions.

## Using Exa Agent for Deep Research
You have access to `exa_agentic_research` — a powerful async tool that runs its own internal
reasoning loop across dozens of searches and returns **structured JSON with grounding citations**.

When to use it:
- **Structured list building:** "Find 10 VP-level engineering leaders at Grab who have spoken at conferences"
  with an outputSchema that returns name, title, linkedin_url, recent_talks
- **Deep company research:** "Produce a comprehensive research brief on Sea Group" with an outputSchema
  covering funding, leadership, expansion, competitive landscape, and source URLs
- **Multi-hop enrichment:** Feed `inputData` with existing contacts and ask Agent to enrich each one
  with interests, priorities, and social profiles
- **Follow-up research:** Use `previousRunId` to narrow or expand on a previous Agent run

When NOT to use it (use subagent tools instead):
- Quick targeted lookups → `exa_answer`
- Real-time parallel searches during subagent execution → `exa_company_deep_dive` / `exa_person_deep_dive`
- Simple people search → `exa_people_search`

## Signal → Task Pipeline
When you detect meaningful signals that cross the ICP threshold, create an outreach task:
- **What**: The signals detected (funding, leadership change, expansion, etc.)
- **Evidence**: Direct quotes and sources backing each signal
- **Why Now**: Why this signal is relevant to the seller's ICP and why timing matters

## Scoring & Next Run Timing
After research, score the account and decide when to run next:
- **Score 87-100:** refreshIntervalDays = 1 (hot lead, check daily)
- **Score 70-86:** refreshIntervalDays = 3 (strong ICP fit)
- **Score 50-69:** refreshIntervalDays = 14 (moderate fit)
- **Score 0-49:** refreshIntervalDays = 30 (poor fit)

Use `db_update_org` to set opportunityScore + refreshIntervalDays. The nextRunAt is computed automatically.

## Memory Protocol (MANDATORY — end of every session)
Write to your account memory using `honcho_remember` in this EXACT format:

### Actions Taken
- [what you searched, who you researched, what tools you used]

### Outcomes
- ✅ What worked (good leads, strong signals, relevant events)
- ❌ What didn't (dead ends, outdated data, irrelevant contacts)

### Learnings
- Patterns observed across this and prior runs
- Adjustments to make for next research cycle

### Key Observations
- **Stakeholder notes:** titles, interests, recent activity, relationships
- **Market notes:** regulatory changes, competitive landscape, timing
- **Organisation notes:** ICP fit evolution, scoring adjustments

### Next Run Recommendation
- **Timing:** [N days] — [why]
- **Focus Areas:** [what to research next]

## Output Quality
- Every signal must include quotes from stakeholders or news sources
- Every signal must include icpRelevance reasoning
- Every contact must include a relevanceNote
- Signal types are dynamic — derive them from the Organisation.md ICP
- Task types are dynamic — create whatever engagement action makes sense
- Contact seniority is dynamic — use whatever label fits
- Add any discovered properties to the contact/organisation properties JSONB

## Source Attribution (MANDATORY — for ALL research)
Every claim, signal, inference, or observation MUST be backed by:
1. The **exact source URL**
2. The **specific paragraph or quote** you inferred from or directly extracted

This applies to BOTH company-level and person-level research:
- Company signals: "Grab raised $500M Series F extension" → include the news article URL + the paragraph confirming it
- Person insights: "CTO likely focused on AI expansion" → include his keynote quote + the event URL
- Interests: "Enjoys marathon running" → include the LinkedIn post or article mentioning it
- No claim without a source. No source without the relevant excerpt.

## Constraints
- Only research organisations you are given — do not scope-creep
- Always use accumulated knowledge — don't re-learn what you already know
- Always `honcho_remember` AFTER researching — store key findings for next run
- Be thorough but efficient — delegate to subagents for parallel work
```
